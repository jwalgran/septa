1 /**
  2  * This module uses the request module as a high level API for consuming the
  3  * SEPTA REST services.
  4  * @private
  5  */
  6 var request = require('request');
  7 
  8 /**
  9  * Make a request to a SEPTA REST API and parse the JSON response.
 10  * @param {String} url A SEPTA REST API url.
 11  * @param {Function} callback The function to call when a response is received
 12  * from the SEPTA API or an error occurs.
 13  * @private
 14  */
 15 var callSeptaApiAndParseJsonResponse = function(url, callback) {
 16     request(url, function (err, response, body) {
 17         if (!err && response.statusCode == 200) {
 18             if (callback) {
 19                 callback(undefined, JSON.parse(body));
 20             }
 21         }
 22         else {
 23             if (callback) {
 24                 callback(err, {"error": {"statusCode": response.statusCode, "body": JSON.parse(body)}});
 25             }
 26         }
 27     });
 28 };
 29 
 30 /**
 31  * Create a new BusRoute
 32  *
 33  * @constructor
 34  * @this {Circle}
 35  * @param {String} number A SEPTA bus route number.
 36  */
 37 var BusRoute = function(number) {
 38     var DETOUR_URL = 'http://www3.septa.org/hackathon/BusDetours',
 39         ALERT_URL = 'http://www3.septa.org/hackathon/Alerts';
 40     this.number = number;
 41     this.detourUrl = DETOUR_URL + '/' + number;
 42     this.alertUrl = ALERT_URL + '/' + number;
 43 }
 44 
 45 /**
 46  * Get the detours for the BusRoute.
 47  *
 48  * @param {function} callback The function to call when a response is received
 49  * from the SEPTA bus detour API or an error occurs. The signature of the callback
 50  * should be:
 51  *
 52  * function(error, response)
 53  *
 54  * The response parameter passed to the callback is an array of objects. Each
 55  * object in the array has three properties: route_direction, reason, and
 56  * current_message. The current_message property can contain embedded newline
 57  * characters. Example:
 58  * [
 59  *   {
 60  *     route_direction: '',
 61  *     reason: '',
 62  *     current_message: 'NORTHBOUND VIA 22ND  \nSTREET\nL - SNYDER\nR - 25TH\nR - WASHINGTON\nL - 22ND \nREGULAR  ROUTE'
 63  *   },
 64  *   {
 65  *     route_direction: '',
 66  *     reason: '',
 67  *     current_message: 'SOUTHBOUND VIA 23RD \nSTREET\nL - WASHINGTON\nR - 21ST\nR - POINT BREEZE\nL - 23  RD\nREGULAR ROUTE'
 68  *   },
 69  *   {
 70  *     route_direction: '',
 71  *     reason: '',
 72  *     current_message: '(Fri, Sat & Sun 6pm - 3am)\nSB via Pennsylvania:\nR - 21St\nReg rte'
 73  *   }
 74  * ]
 75  */
 76 BusRoute.prototype.getDetours = function(callback) {
 77     callSeptaApiAndParseJsonResponse(this.detourUrl, function(err, resp) {
 78         if (!err){
 79             callback(undefined, resp[0].route_info);
 80         } else {
 81             callback(err, resp);
 82         }
 83     });
 84 }
 85 
 86 /**
 87  * Get the current alerts for the BusRoute.
 88  *
 89  * @param {function} callback The function to call when a response is received
 90  * from the SEPTA alert API or an error occurs. The signature of the callback
 91  * should be:
 92  *
 93  * function(error, response)
 94  *
 95  * The response parameter passed to the callback is a string containing the
 96  * current alerts for the BusRoute.
 97  */
 98 BusRoute.prototype.getAlerts = function(callback) {
 99     callSeptaApiAndParseJsonResponse(this.alertUrl, function(err, resp) {
100         if (!err){
101             callback(undefined, resp[0].current_message);
102         } else {
103             callback(err, resp);
104         }
105     });
106 }
107 
108 exports.BusRoute = BusRoute;
